package io.github.prolobjectlink.prolog.tuprolog;

import alice.tuprolog.Var;
import io.github.prolobjectlink.prolog.PrologProvider;
import io.github.prolobjectlink.prolog.PrologResult;
import io.github.prolobjectlink.prolog.PrologTerm;
import io.github.prolobjectlink.prolog.PrologTermType;
import io.github.prolobjectlink.prolog.PrologVariable;

public class TuPrologResult extends TuPrologVariable implements PrologResult {

	TuPrologResult(PrologProvider provider, String name) {
		super(PrologTermType.RESULT_TYPE, provider, name);
	}

	TuPrologResult(PrologProvider provider, PrologTerm name) {
		super(PrologTermType.RESULT_TYPE, provider);
		this.value = new Var(((PrologVariable) name).getName());
	}

	TuPrologResult(PrologProvider provider, int position) {
		super(PrologTermType.RESULT_TYPE, provider, new Var());
	}

	TuPrologResult(PrologProvider provider, String name, int position) {
		super(PrologTermType.RESULT_TYPE, provider, new Var(name));
	}

	public final PrologTerm getNameTerm() {
		return provider.newVariable(((Var) value).getName(), 0);
	}

	public final String getName() {
		return ((Var) value).getName();
	}
}
